        {
            int         c, i;

            for ( i = 0;  (c = getchar()) != EOF && c != '\n' && i < size - 1; i++ ) {
                s[i] = c;
            }

            s[i] = '\0';

            if ( c == EOF && i == 0 ){
                return -1;
            }

            return i;

            #include <stdio.h>
            #include <math.h>
            int main(void)
            {
                double a, b, c, D;
                printf("2次の方程式ax^2+bx+c=0の解を求めます。\na,b,c,の値を入力してください。\na：");
                scanf("%lf", &a);
                printf("b；");
                scanf("%lf", &b);
                printf("c；");
                scanf("%lf", &c);
                
                if (a == 0){
                    // 二次関数でない場合
                    if (b == 0 && c == 0){
                        // a,b,cがすべて0の場合
                        printf("解は任意の数");
                    }else if (b == 0){
                        // a,bが0の場合
                        printf("解なし");
                    }else{
                        // aのみ0の場合
                        printf("解は%fです。", -c / b);
                    }
                }else{
                    // 二次関数の場合
                    D = (b * b - 4 * a * c); // 判別式
                    if (D > 0){
                        // 実数解を2つ持つ場合
                        printf("解は%fと%fです。", (-b + sqrt(D)) / (2 * a), (-b - sqrt(D)) / (2 * a));
                    }else if (D == 0){
                        // 重解をもつ場合
                        printf("解は%fです。", (-b) / (2 * a));
                    }else{
                        // 虚数解を持つ場合
                        printf("解は%f±√%fiです。", (-b) / (2 * a), -D / (4 * a * a));
                    }
                }
                
                return 0;
            }

半径１の円に内接する，2のn乗（ 2 ≦ n )正多角形は，n の値が大きくなると円に近づく
正多角形の辺の長さを計算することで，円周率の近似値を算出するプログラムを作成せよ．


まずは，正方形から考えてみよう．
まず，xy座標上で，x軸，y軸上に頂点を持つような正方形を考える．
この時の，各頂点の座標は，(1,0), (0,1), (-1,0), (0,-1)となる．
各辺の長さは √2 なので，全部の辺の長さは 4√2である． 従って，2π（円周の長さ）の近似値は 4√2 と考えられる．

次に，正方形と頂点を共有する正八角形の頂点を求めることを考える．
ただし，正多角形なので，１辺の長ささえ分かれば全辺の和を計算することができるので，(1,0)と(0,1)の間にできる新たな頂点をのみを考える．
(1,0)と(0,1) の中点は，(0.5, 0.5)である．
原点から(0.5, 0.5)までの距離は √2/2なので，(0.5, 0.5)を√2/2で割ると， 正8角形の新たな頂点(√2/2,√2/2) が求まり，(1,0）からの距離として 正8角形の1辺の長さを算出することが可能となる．

              #include <stdio.h>
              #include <math.h>

              int main(void)
              {
                  double Npolygon, vertexX, vertexY, mid_pointX, mid_pointY, mid_length, ans_vertexX, ans_vertexY, polygon_length, ans_pi;
                  int n = 24;

                  printf("正多角形から円周率を近似します。\n数字が変化しなくなる正2^24角形まで近似します。\n");
                  // 正方形の時の値を代入
                  // 正方形の(1,0)ではない頂点の値を入れる
                  vertexX = 0;
                  vertexY = 1;

                  // 初回は4角形から始める
                  Npolygon = 4;

                  for (int i = 1; i <= n; i++)
                  {
                      // 中点を求める
                      mid_pointX = (vertexX + 1.0) / 2.0;
                      mid_pointY = vertexY / 2.0;
                      // 原点と中点の長さを求める
                      mid_length = sqrt(pow(mid_pointX, 2.0) + pow(mid_pointY, 2.0));
                      // 原点と中点を通る直線上で、原点との長さが1の点を求める
                      ans_vertexX = mid_pointX / mid_length;
                      ans_vertexY = mid_pointY / mid_length;
                      // 求めた点と(1,0)との長さが、多角形の1辺の長さになる
                      polygon_length = sqrt(pow((1 - ans_vertexX), 2.0) + pow(ans_vertexY, 2.0));
                      // 近似の円周率を求める
                      ans_pi = (polygon_length * Npolygon);

                      // 近似で出した円周率を小数14桁まで表示（double型の精度が15桁までだから
                      printf("%8d角形での近似 π = %.14f\n", (int)Npolygon, ans_pi);
                      // 次の頂点の値を変更
                      vertexX = ans_vertexX;
                      vertexY = ans_vertexY;
                      // 次のn角形のnを2倍する
                      Npolygon *= 2;
                  }
                  printf("近似で求めた円周率   π = %.14f \n", ans_pi);
                  return 0;
              }
